  ```typescript
  let q = supabaseAdmin
    .from('admin_users_view')
    .select('id, first_name, last_name, phone, email, last_sign_in_at, banned_until, created_at', { count: 'exact' })

  if (trimmed) {
    const term = `%${trimmed}%`
    q = q.or(`first_name.ilike.${term},last_name.ilike.${term},email.ilike.${term},phone.ilike.${term}`)
  }

  const { data, count, error } = await q
    .order('created_at', { ascending: false })
    .range(offset, offset + 24) // 25 per page
  ```
  Then batch-fetch memberships for just the returned user IDs (same pattern as current enrichProfilesWithAuthAndMembership but WITHOUT calling listUsers).

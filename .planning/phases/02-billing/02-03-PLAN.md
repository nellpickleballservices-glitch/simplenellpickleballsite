Create `app/[locale]/(member)/checkout-success/page.tsx` — this is a Client Component (requires 'use client' for useEffect and useState). The Supabase Realtime subscription must run in useEffect:

```typescript
'use client'

import { useEffect, useState } from 'react'
import { useTranslations } from 'next-intl'
import Link from 'next/link'
import { createBrowserClient } from '@supabase/ssr'

type MembershipStatus = 'pending' | 'active' | 'timeout'

export default function CheckoutSuccessPage() {
  const t = useTranslations('Billing.checkoutSuccess')
  const [status, setStatus] = useState<MembershipStatus>('pending')

  useEffect(() => {
    const supabase = createBrowserClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY!
    )

    let channel: ReturnType<typeof supabase.channel> | null = null
    let timeoutId: ReturnType<typeof setTimeout> | null = null

    supabase.auth.getUser().then(({ data: { user } }) => {
      if (!user) return

      // 30-second timeout — webhook may legitimately take this long under load.
      // The timeout fallback tells the user payment was received without showing an error.
      timeoutId = setTimeout(() => setStatus('timeout'), 30_000)

      channel = supabase
        .channel('membership-status')
        .on(
          'postgres_changes',
          {
            event: 'UPDATE',
            schema: 'public',
            table: 'memberships',
            filter: `user_id=eq.${user.id}`,
          },
          (payload) => {
            if ((payload.new as { status: string }).status === 'active') {
              if (timeoutId) clearTimeout(timeoutId)
              setStatus('active')
              if (channel) supabase.removeChannel(channel)
            }
          }
        )
        .subscribe()
    })

    return () => {
      if (timeoutId) clearTimeout(timeoutId)
      if (channel) supabase.removeChannel(channel)
    }
  }, [])

  return (
    <main className="min-h-screen flex flex-col items-center justify-center p-8 text-center">
      {status === 'pending' && (
        <div>
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-black mx-auto mb-4" />
          <p className="text-lg">{t('pending')}</p>
        </div>
      )}
      {status === 'active' && (
        <div>
          <h1 className="text-3xl font-bold mb-2">{t('active')}</h1>
          <p className="text-gray-600 mb-6">{t('activeSubtitle')}</p>
          <Link href="/es/dashboard" className="bg-black text-white py-2 px-6 rounded hover:bg-gray-800">
            {t('goToDashboard')}
          </Link>
        </div>
      )}
      {status === 'timeout' && (
        <div>
          <h1 className="text-2xl font-bold mb-2">{t('timeout')}</h1>
          <p className="text-gray-600 mb-6">{t('timeoutSubtitle')}</p>
          <Link href="/es/dashboard" className="text-blue-600 hover:underline">
            {t('goToDashboard')}
          </Link>
        </div>
      )}
    </main>
  )
}
```

NOTE: This is a Client Component but uses useTranslations. This is correct — next-intl 4 supports useTranslations in both Server and Client Components when wrapped by NextIntlClientProvider (already in layout.tsx from Plan 01-04).

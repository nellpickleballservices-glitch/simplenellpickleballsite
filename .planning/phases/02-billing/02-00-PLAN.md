```sql
-- NELL Pickleball Club — Phase 2 Billing Schema Additions
-- Run in: Supabase Dashboard -> SQL Editor AFTER 0001_initial_schema.sql
-- Adds: webhook_events idempotency table, memberships UNIQUE(user_id) constraint,
--       supabase_realtime publication for memberships table.

-- ============================================================
-- 1. webhook_events — exactly-once processing for Stripe events
-- ============================================================
-- UNIQUE on stripe_event_id is the idempotency guard.
-- Webhook handler inserts this row BEFORE processing any business logic.
-- On Postgres error 23505 (unique_violation), the event was already processed.
CREATE TABLE webhook_events (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  stripe_event_id TEXT NOT NULL UNIQUE,
  event_type      TEXT NOT NULL,
  processed_at    TIMESTAMPTZ DEFAULT NOW()
);
-- Enable RLS per project policy (only accessed by service_role in webhook handler)
ALTER TABLE webhook_events ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Service role full access on webhook_events"
  ON webhook_events FOR ALL TO service_role USING (true) WITH CHECK (true);

-- ============================================================
-- 2. memberships UNIQUE(user_id) constraint
-- ============================================================
-- Required for supabase upsert({ onConflict: 'user_id' }) to work correctly.
-- One active membership row per user. Prevents duplicate rows if checkout fires twice.
ALTER TABLE memberships ADD CONSTRAINT memberships_user_id_key UNIQUE (user_id);

-- ============================================================
-- 3. Enable Supabase Realtime for memberships table
-- ============================================================
-- Without this, postgres_changes subscriptions on memberships never fire.
-- The post-checkout page polls this table via Supabase Realtime.
-- If this is already in the publication (e.g., toggled via Dashboard), this is a no-op.
ALTER PUBLICATION supabase_realtime ADD TABLE memberships;
```
